# Preprocessing: Load images, apply CLAHE and Laplacian filter
def apply_clahe(image):
    lab = cv2.cvtColor(image, cv2.COLOR_BGR2Lab)
    l, a, b = cv2.split(lab)
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    cl = clahe.apply(l)
    limg = cv2.merge((cl, a, b))
    return cv2.cvtColor(limg, cv2.COLOR_Lab2BGR)

def apply_laplacian_filter(image):
    laplacian = cv2.Laplacian(image, cv2.CV_64F)
    sharpened = np.uint8(np.clip(image - laplacian, 0, 255))
    return sharpened

def preprocess_image(img_path):
    try:
        img = load_img(img_path, target_size=(224, 224))
        img_array = img_to_array(img)
        img_array = apply_clahe(img_array.astype('uint8'))
        img_array = apply_laplacian_filter(img_array)
        return img_array
    except UnidentifiedImageError:
        print(f"Skipping file {img_path} - Unidentified image format.")
        return None

def load_and_preprocess_images(data_dir, classes):
    images = []
    labels = []
    
    for label, class_name in enumerate(classes):
        folder = os.path.join(data_dir, class_name)
        print(f"Looking for images in: {folder}")  # Debugging: print the folder path
        if not os.path.exists(folder):
            print(f"Error: The folder {folder} does not exist.")
            continue  # Skip if the folder doesn't exist
        for img_name in os.listdir(folder):
            img_path = os.path.join(folder, img_name)
            if img_name.lower().endswith(('.png', '.jpg', '.jpeg')):  # Check for image formats
                img_array = preprocess_image(img_path)
                if img_array is not None:  # Only append if preprocessing was successful
                    images.append(img_array)
                    labels.append(label)
            else:
                print(f"Skipping file {img_path} - Unsupported format.")
    
    images = np.array(images)
    labels = np.array(labels)
    
    return images, labels

# Compute class-wise metrics
def compute_classwise_metrics(y_true, y_pred, y_prob, num_classes=2):
    cm = confusion_matrix(y_true, y_pred)
    classwise_accuracy = cm.diagonal() / cm.sum(axis=1)
    
    classwise_auc = []
    for i in range(num_classes):
        # Create binary labels for the current class
        y_true_binary = (y_true == i).astype(int)
        y_prob_binary = y_prob[:, i] if num_classes > 2 else y_prob
        
        # Compute ROC curve and AUC
        fpr, tpr, _ = roc_curve(y_true_binary, y_prob_binary)
        auc_value = auc(fpr, tpr)
        classwise_auc.append(auc_value)
    
    return classwise_accuracy, classwise_auc, cm
